-- Create the 'cat' table with the specified columns and constraints
CREATE TABLE cat (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    description TEXT NOT NULL,
    breed VARCHAR(255) NOT NULL,
    image_url TEXT NOT NULL,
    api_id VARCHAR(50) UNIQUE,
    is_favorite BOOLEAN DEFAULT FALSE
);

-- Create the 'favorite_cat' table with a foreign key reference to 'cat' table
CREATE TABLE favorite_cat (
    id SERIAL PRIMARY KEY,
    cat_id INTEGER NOT NULL,
    FOREIGN KEY (cat_id) REFERENCES cat(id) ON DELETE CASCADE
);
